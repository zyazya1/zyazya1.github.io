# 1.4.5.2 变量定义

本模块用于定义通用变量，通常变量定义好之后不再修改，否则重新导入程序后，变量值会清零。

```
pers num p;
pers num d;
pers num z;
pers num yz{4,8};
pers num yx{4,8};
pers num bz{4,8};
pers num bx{4,8};
pers num fz{4,8};
pers num fx{4,8};
pers num lx{4,5}:=[[1,2,3,3,4],[1,2,3,3,4],[1,2,3,4,4],[1,2,2,3,4]];
pers num bgw{4}:=[];
pers num gwb{4}:=[];
pers num k{4}:=[4,8,7,7];
pers num jsq{50};
pers num bh{4}:=[1,1,1,1];
pers num ygb{4};
pers num r1;
pers num r2;
var robtarget d1;

var socketdev s1;
var byte fs{30};
var byte jsh{30};

var socketdev s2;
var bool ok;
pers string sc{4}:=["s 1","s 2","s 3","s 4"];
var string sr;

var intnum a4;
var intnum a5;
var intnum a6;

proc ql()
	for i from 1 to 4 do
		for j from 1 to 8 do
			yz{i,j}:=0;
			yx{i,j}:=0;
			bz{i,j}:=0;
			bx{i,j}:=0;
			fz{i,j}:=0;
			fx{i,j}:=0;
		endfor
	endfor
endproc

proc jsq0(num a,num b)
	for i from a to b do
		jsq{i}:=0;
	endfor
endproc
```
